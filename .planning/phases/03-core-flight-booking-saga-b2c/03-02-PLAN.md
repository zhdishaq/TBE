<objective>
Isolate all payment concerns inside PaymentService per D-10/PAY-08 (PCI SAQ-A boundary). Build the Stripe Payment Intent flow (authorize → capture → cancel → refund) with deterministic idempotency keys per D-13, the webhook ingress with signature verification per D-12, and the B2B credit-wallet implementation using Dapper with `UPDLOCK, ROWLOCK, HOLDLOCK` per D-14/D-15. Publish payment + wallet lifecycle events back to the saga via the outbox.

This plan does NOT touch BookingService. It subscribes to saga commands (defined in 03-01) and emits saga-consumed events. It runs in Wave 2 after 03-01's contracts + saga skeleton land, in parallel with 03-04 (notifications) which touches disjoint files.

Purpose: Provides the only service in the solution that talks to Stripe or mutates the wallet ledger. Every other service accesses payment state exclusively via TBE.Contracts messages.

Output: IStripePaymentGateway + StripePaymentGateway, five command consumers (Authorize/Capture/Cancel/Refund + three wallet consumers), StripeWebhookController with signature verification, Dapper-based WalletRepository, PaymentDbContext extended with WalletTransactions + StripeWebhookEvents, migration, WalletController (admin read-only view for PAY-03), and unit tests including a concurrency test for wallet double-spend.
</objective>

    ```csharp
    [ApiController] [Route("webhooks/stripe")] [AllowAnonymous]
    public sealed class StripeWebhookController(
        IOptionsMonitor<StripeOptions> opts,
        PaymentDbContext db,
        IPublishEndpoint publish,
        ILogger<StripeWebhookController> log) : ControllerBase
    {
        [HttpPost]
        public async Task<IActionResult> HandleAsync(CancellationToken ct)
        {
            using var reader = new StreamReader(Request.Body);
            var json = await reader.ReadToEndAsync(ct);
            var sig = Request.Headers["Stripe-Signature"].ToString();
            Stripe.Event evt;
            try { evt = EventUtility.ConstructEvent(json, sig, opts.CurrentValue.WebhookSecret, tolerance: 300); }
            catch (StripeException) { return BadRequest(); }

            if (await db.StripeWebhookEvents.AsNoTracking().AnyAsync(e => e.EventId == evt.Id, ct))
                return Ok();

            // Extract booking_id from metadata (only present on payment_intent.* events)
            string? paymentIntentId = null; Guid? bookingId = null;
            if (evt.Data.Object is Stripe.PaymentIntent pi) {
                paymentIntentId = pi.Id;
                if (pi.Metadata != null && pi.Metadata.TryGetValue("booking_id", out var bid) && Guid.TryParse(bid, out var g))
                    bookingId = g;
            }

            db.StripeWebhookEvents.Add(new StripeWebhookEvent { EventId = evt.Id, EventType = evt.Type, ReceivedAtUtc = DateTime.UtcNow });
            await publish.Publish(new StripeWebhookReceived(evt.Id, evt.Type, paymentIntentId, DateTimeOffset.UtcNow), ct);
            // Also publish a typed saga event directly when booking_id resolvable — reduces one RabbitMQ hop:
            if (bookingId is { } b) {
                if (evt.Type == "payment_intent.amount_capturable_updated") await publish.Publish(new PaymentAuthorized(b, paymentIntentId!, DateTimeOffset.UtcNow), ct);
                else if (evt.Type == "payment_intent.succeeded") await publish.Publish(new PaymentCaptured(b, paymentIntentId!, DateTimeOffset.UtcNow), ct);
                else if (evt.Type == "payment_intent.payment_failed") await publish.Publish(new PaymentAuthorizationFailed(b, "stripe webhook: payment_failed"), ct);
            }
            await db.SaveChangesAsync(ct);  // outbox captures the publishes atomically
            return Ok();
        }
    }
    ```
    This controller MUST NOT log `json`, `sig`, `pi`, or `evt` contents (Pitfall 7).

    ```csharp
    public sealed class WalletRepository(IConfiguration cfg, ILogger<WalletRepository> log) : IWalletRepository
    {
        private string Cs => cfg.GetConnectionString("PaymentDb")!;

        public async Task<Guid> ReserveAsync(Guid walletId, Guid bookingId, decimal amount, string currency, CancellationToken ct)
        {
            using var conn = new SqlConnection(Cs);
            await conn.OpenAsync(ct);
            using var tx = (SqlTransaction)await conn.BeginTransactionAsync(IsolationLevel.ReadCommitted, ct);

            var available = await conn.ExecuteScalarAsync<decimal>(
                new CommandDefinition(
                    @"SELECT ISNULL(SUM(SignedAmount), 0) FROM payment.WalletTransactions
                      WITH (UPDLOCK, ROWLOCK, HOLDLOCK) WHERE WalletId = @WalletId",
                    new { WalletId = walletId }, tx, cancellationToken: ct));

            if (available < amount)
                throw new InsufficientWalletBalanceException(walletId, available, amount);

            var txId = Guid.NewGuid();
            await conn.ExecuteAsync(new CommandDefinition(
                @"INSERT INTO payment.WalletTransactions (TxId, WalletId, BookingId, EntryType, Amount, Currency, IdempotencyKey)
                  VALUES (@TxId, @WalletId, @BookingId, 1, @Amount, @Currency, @IdemKey)",
                new { TxId = txId, WalletId = walletId, BookingId = bookingId, Amount = amount, Currency = currency,
                      IdemKey = $"booking-{bookingId}-reserve" }, tx, cancellationToken: ct));

            await tx.CommitAsync(ct);
            return txId;
        }
        // Commit, Release, TopUp, GetBalance, List implementations analogous.
    }
    ```
