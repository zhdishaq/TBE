<objective>
Isolate all payment concerns inside PaymentService per D-10/PAY-08 (PCI SAQ-A boundary). Build the Stripe Payment Intent flow (authorize → capture → cancel → refund) with deterministic idempotency keys per D-13, the webhook ingress with signature verification per D-12, the B2B credit-wallet implementation using Dapper with `UPDLOCK, ROWLOCK, HOLDLOCK` per D-14/D-15, AND the PAY-04 wallet top-up flow via Stripe. Publish payment + wallet lifecycle events back to the saga via the outbox.

**Webhook architecture (W3 clarification):** `StripeWebhookController` is a dumb ingress — it verifies the signature, dedups by event.Id, and publishes EXACTLY ONE typed envelope (`StripeWebhookReceived`). It does NOT publish saga events directly. `StripeWebhookConsumer` is the ONLY publisher of `PaymentAuthorized` / `PaymentCaptured` / `PaymentAuthorizationFailed`. Similarly `StripeTopUpConsumer` is the ONLY writer of TopUp ledger entries. This preserves a single translation boundary and makes testing deterministic.

This plan does NOT touch BookingService. It subscribes to saga commands (defined in 03-01) and emits saga-consumed events. It runs in Wave 2 after 03-01's contracts + saga skeleton land, in parallel with 03-04 (notifications) which touches disjoint files.

Purpose: Provides the only service in the solution that talks to Stripe or mutates the wallet ledger. Every other service accesses payment state exclusively via TBE.Contracts messages.

Output: IStripePaymentGateway + StripePaymentGateway, five command consumers (Authorize/Capture/Cancel/Refund + three wallet consumers), StripeWebhookController with signature verification (ingress only), StripeWebhookConsumer (sole saga-event publisher), StripeTopUpConsumer (sole TopUp writer), Dapper-based WalletRepository, PaymentDbContext extended with WalletTransactions + StripeWebhookEvents, migration, WalletController (admin read-only view for PAY-03 + top-up endpoint for PAY-04), and unit tests including a concurrency test for wallet double-spend.
</objective>

    ```csharp
    [ApiController] [Route("webhooks/stripe")] [AllowAnonymous]
    public sealed class StripeWebhookController(
        IOptionsMonitor<StripeOptions> opts,
        PaymentDbContext db,
        IPublishEndpoint publish,
        ILogger<StripeWebhookController> log) : ControllerBase
    {
        [HttpPost]
        public async Task<IActionResult> HandleAsync(CancellationToken ct)
        {
            using var reader = new StreamReader(Request.Body);
            var json = await reader.ReadToEndAsync(ct);
            var sig = Request.Headers["Stripe-Signature"].ToString();
            Stripe.Event evt;
            try { evt = EventUtility.ConstructEvent(json, sig, opts.CurrentValue.WebhookSecret, tolerance: 300); }
            catch (StripeException) { return BadRequest(); }

            if (await db.StripeWebhookEvents.AsNoTracking().AnyAsync(e => e.EventId == evt.Id, ct))
                return Ok();

            string? paymentIntentId = null;
            Guid? bookingId = null, walletId = null, agencyId = null;
            decimal? topUpAmount = null;
            if (evt.Data.Object is Stripe.PaymentIntent pi)
            {
                paymentIntentId = pi.Id;
                if (pi.Metadata is { } md)
                {
                    if (md.TryGetValue("booking_id", out var b) && Guid.TryParse(b, out var bg)) bookingId = bg;
                    if (md.TryGetValue("wallet_id",  out var w) && Guid.TryParse(w, out var wg)) walletId  = wg;
                    if (md.TryGetValue("agency_id",  out var a) && Guid.TryParse(a, out var ag)) agencyId  = ag;
                    if (md.TryGetValue("topup_amount", out var t) && decimal.TryParse(t, out var td)) topUpAmount = td;
                }
            }

            db.StripeWebhookEvents.Add(new StripeWebhookEvent { EventId = evt.Id, EventType = evt.Type, ReceivedAtUtc = DateTime.UtcNow });
            await publish.Publish(new StripeWebhookReceived(
                evt.Id, evt.Type, paymentIntentId, bookingId, walletId, topUpAmount, agencyId, DateTimeOffset.UtcNow), ct);
            await db.SaveChangesAsync(ct);  // outbox captures the single publish atomically
            return Ok();
        }
    }
    ```

    public sealed class StripeTopUpConsumer(
        IWalletRepository wallet,
        ILogger<StripeTopUpConsumer> log) : IConsumer<StripeWebhookReceived>
    {
        private static readonly HashSet<string> TopUpEventTypes = new()
        {
            "charge.succeeded",
            "payment_intent.succeeded"
        };

        public async Task Consume(ConsumeContext<StripeWebhookReceived> ctx)
        {
            var e = ctx.Message;
            if (e.WalletId is not { } walletId) return;                 // not a top-up envelope
            if (!TopUpEventTypes.Contains(e.EventType)) return;
            if (e.TopUpAmount is not { } amountCents || amountCents <= 0) {
                log.LogWarning("top-up envelope for wallet {WalletId} missing topup_amount; event {EventId}", walletId, e.EventId);
                return;
            }
            if (string.IsNullOrWhiteSpace(e.PaymentIntentId)) {
                log.LogWarning("top-up envelope for wallet {WalletId} missing PaymentIntentId; event {EventId}", walletId, e.EventId);
                return;
            }

            var idemKey = $"wallet-{walletId}-topup-{e.PaymentIntentId}";
            var currency = "GBP"; // v1 fixed; multi-currency top-up is Phase 6 scope

            try
            {
                var txId = await wallet.TopUpAsync(walletId, amountCents / 100m, currency, idemKey, ctx.CancellationToken);
                await ctx.Publish(new WalletToppedUp(walletId, e.AgencyId, amountCents / 100m, currency, e.PaymentIntentId!, DateTimeOffset.UtcNow));
                log.LogInformation("wallet {WalletId} topped up {Amount} {Currency} (tx {TxId})", walletId, amountCents / 100m, currency, txId);
            }
            catch (Exception ex) when (IsUniqueViolation(ex))
            {
                log.LogInformation("wallet top-up replay suppressed for {IdemKey}", idemKey);
            }
        }

        private static bool IsUniqueViolation(Exception ex) =>
            ex is Microsoft.Data.SqlClient.SqlException sql && (sql.Number == 2601 || sql.Number == 2627)
            || ex.InnerException is Microsoft.Data.SqlClient.SqlException inner && (inner.Number == 2601 || inner.Number == 2627);
    }
