```sql
;WITH Src AS (
    SELECT CAST(InitiatedAtUtc AS date) AS Date,
           COALESCE(ProductType, 'Flight') AS Product,   -- defaulted in v1
           CASE Channel WHEN 0 THEN 'B2C' WHEN 1 THEN 'B2B' WHEN 2 THEN 'Manual' END AS Channel,
           COUNT(*) AS BookingsCount,
           SUM(GrossAmount) AS Revenue,
           SUM(COALESCE(AgencyCommissionAmount, 0)) AS Commission,
           SUM(BaseFareAmount) AS NetAmount,
           SUM(COALESCE(MarkupAmount, 0)) AS MarkupAmount
    FROM dbo.BookingSagaState
    WHERE Status = 'Confirmed'
      AND InitiatedAtUtc >= @FromUtc AND InitiatedAtUtc < @ToUtcExclusive
    GROUP BY CAST(InitiatedAtUtc AS date),
             COALESCE(ProductType, 'Flight'),
             CASE Channel WHEN 0 THEN 'B2C' WHEN 1 THEN 'B2B' WHEN 2 THEN 'Manual' END
)
MERGE reporting.MisDailyAggregates WITH (HOLDLOCK) AS tgt
USING Src AS src ON tgt.Date = src.Date AND tgt.Product = src.Product AND tgt.Channel = src.Channel
WHEN MATCHED THEN UPDATE SET
    BookingsCount = src.BookingsCount,
    Revenue = src.Revenue,
    Commission = src.Commission,
    NetAmount = src.NetAmount,
    MarkupAmount = src.MarkupAmount,
    LastAggregatedAt = SYSUTCDATETIME()
WHEN NOT MATCHED BY TARGET THEN INSERT (Date, Product, Channel, BookingsCount, Revenue, Commission, NetAmount, MarkupAmount, LastAggregatedAt)
VALUES (src.Date, src.Product, src.Channel, src.BookingsCount, src.Revenue, src.Commission, src.NetAmount, src.MarkupAmount, SYSUTCDATETIME());

INSERT INTO payment.CommissionAccruals (Id, BookingId, AgencyId, GrossAmount, NetAmount, MarkupAmount, CommissionAmount, Currency, BookingConfirmedAt, PeriodKey, Status, CreatedAt)
SELECT NEWID(), b.CorrelationId, b.AgencyId, b.GrossAmount, b.BaseFareAmount, COALESCE(b.MarkupAmount,0),
       COALESCE(b.AgencyCommissionAmount, 0), b.Currency, b.ConfirmedAtUtc,
       FORMAT(b.ConfirmedAtUtc, 'yyyy-MM'), 'Draft', SYSUTCDATETIME()
FROM dbo.BookingSagaState b
LEFT JOIN payment.CommissionAccruals a ON a.BookingId = b.CorrelationId
WHERE b.Status = 'Confirmed' AND b.AgencyId IS NOT NULL
  AND b.ConfirmedAtUtc >= @FromUtc AND b.ConfirmedAtUtc < @ToUtc
  AND b.AgencyCommissionAmount > 0
  AND a.Id IS NULL;

// Only run if clock.Today is first business day (Mon-Fri) of 1-3 of month — caller checks
var groups = await _db.CommissionAccruals
    .Where(a => a.PeriodKey == periodKey && a.Status == "Draft")
    .GroupBy(a => a.AgencyId)
    .Select(g => new { AgencyId = g.Key, Accruals = g.ToList(),
                       Total = g.Sum(x => x.CommissionAmount),
                       Gross = g.Sum(x => x.GrossAmount),
                       Net = g.Sum(x => x.NetAmount),
                       Markup = g.Sum(x => x.MarkupAmount),
                       Count = g.Count(),
                       Currency = g.First().Currency })
    .ToListAsync(ct);

int generated = 0;
foreach (var g in groups) {
    // idempotency pre-check
    var exists = await _db.AgencyMonthlyStatements
        .AnyAsync(s => s.AgencyId == g.AgencyId && s.PeriodKey == periodKey, ct);
    if (exists) continue;

    var stmt = new AgencyMonthlyStatementRow {
        Id = Guid.NewGuid(), AgencyId = g.AgencyId, PeriodKey = periodKey,
        AccrualCount = g.Count, TotalCommission = g.Total, TotalGross = g.Gross,
        TotalNet = g.Net, TotalMarkup = g.Markup, Currency = g.Currency,
        Status = "Draft", GeneratedAt = DateTime.UtcNow
    };
    stmt.PdfBytes = await _pdfGenerator.GenerateAsync(stmt, g.Accruals, ct);
    _db.AgencyMonthlyStatements.Add(stmt);

    var accrualIds = g.Accruals.Select(a => a.Id).ToList();
    await _db.CommissionAccruals
        .Where(a => accrualIds.Contains(a.Id))
        .ExecuteUpdateAsync(s =>
            s.SetProperty(a => a.Status, "Statemented")
             .SetProperty(a => a.StatementId, stmt.Id), ct);

    await _db.SaveChangesAsync(ct);   // commits statement insert + ExecuteUpdate already committed; wrap in Database.BeginTransaction for atomicity
    generated++;
}
return generated;
```
